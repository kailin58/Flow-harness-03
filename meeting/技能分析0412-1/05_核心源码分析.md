```javascript
  'google.com', 'youtube.com', 'android.com', 'chromium.org',

];

const BRAND_FONT_DOMAINS = {
  'roboto': GOOGLE_DOMAINS,
  'google sans': GOOGLE_DOMAINS,
  'product sans': GOOGLE_DOMAINS,
};

function isBrandFontOnOwnDomain(font) {
  if (typeof location === 'undefined') return false;
  const allowed = BRAND_FONT_DOMAINS[font];
  if (!allowed) return false;
  const host = location.hostname.toLowerCase();
  return allowed.some(suffix => host === suffix || host.endsWith('.' + suffix));
}
```

**智能检测**: 在品牌自己的网站上使用品牌字体不算"过度使用"

## 二、颜色工具函数

### 2.1 中性色判断

```javascript
function isNeutralColor(color) {
  if (!color || color === 'transparent') return true;

  // RGB/RGBA - 使用通道差值
  const rgb = color.match(/rgba?\((\d+),\s*(\d+),\s*(\d+)/);
  if (rgb) {
    return (Math.max(+rgb[1], +rgb[2], +rgb[3]) - 
            Math.min(+rgb[1], +rgb[2], +rgb[3])) < 30;
  }

  // OKLCH/LCH - chroma 是第二个数值分量
  const oklch = color.match(/oklch\(\s*[\d.%-]+\s+([\d.-]+)/i);
  if (oklch) return parseFloat(oklch[1]) < 0.02;
  
  // ... 其他格式处理
}
```

**阈值说明**:
- RGB: 通道差值 < 30 ≈ 11.7% 的 0-255 范围
- OKLCH: chroma < 0.02 读取为有色调，不是灰色
- LCH: chroma < 3 读取为有色调

### 2.2 相对亮度计算

```javascript
function relativeLuminance({ r, g, b }) {
  const [rs, gs, bs] = [r / 255, g / 255, b / 255].map(c =>
    c <= 0.03928 ? c / 12.92 : ((c + 0.055) / 1.055) ** 2.4
  );
  return 0.2126 * rs + 0.7156 * gs + 0.0722 * bs;
}

function contrastRatio(c1, c2) {
  const l1 = relativeLuminance(c1);
  const l2 = relativeLuminance(c2);
  return (Math.max(l1, l2) + 0.05) / (Math.min(l1, l2) + 0.05);
}
```

**符合 WCAG 标准**: 使用 sRGB 非线性变换

## 四、关键检测函数示例

### 4.1 边框检测

```javascript
function checkBorders(opts) {
  const { tag, style, rect, width } = opts;
  
  if (BORDER_SAFE_TAGS.has(tag)) return [];
  
  const findings = [];
  
  // 解析边框属性
  const borders = [
    { side: 'left', width: style.borderLeftWidth, color: style.borderLeftColor },
    { side: 'right', width: style.borderRightWidth, color: style.borderRightColor },
    { side: 'top', width: style.borderTopWidth, color: style.borderTopColor },
    { side: 'bottom', width: style.borderBottomWidth, color: style.borderBottomColor },
  ];
  
  // 检测侧条纹边框
  const activeBorders = borders.filter(b => 
    parseFloat(b.width) >= 2 && 
    b.color !== 'transparent' &&
    !isNeutralColor(b.color)
  );
  
  // 单侧粗边框 = side-tab
  if (activeBorders.length === 1) {
    findings.push({
      id: 'side-tab',
      snippet: `${activeBorders[0].side}-tab on "${getElementContext(el)}"`
    });
  }
  
  return findings;
}
```

## 五、页面级检测

### 5.1 扁平类型层次检测

```javascript
function checkPageTypography(document) {
  const findings = [];
  
  // 收集所有字体大小
  const sizes = new Map();
  for (const el of document.querySelectorAll('h1, h2, h3, h4, h5, h6, p, .heading')) {
    const size = parseFloat(getComputedStyle(el).fontSize);
    if (!sizes.has(size)) sizes.set(size, []);
    sizes.get(size).push(el);
  }
  
  // 检查比例
  const sortedSizes = [...sizes.keys()].sort((a, b) => b - a);
  
  for (let i = 0; i < sortedSizes.length - 1; i++) {
    const ratio = sortedSizes[i] / sortedSizes[i + 1];
    if (ratio < 1.25) {
      // 相邻大小太接近
      findings.push({
        id: 'flat-type-hierarchy',
        snippet: `ratio ${ratio.toFixed(2)} between ${sortedSizes[i]}px and ${sortedSizes[i + 1]}px`
      });
    }
  }
  
  return findings;
}
```

### 5.2 单一字体检测

```javascript
function checkFonts(document) {
  const findings = [];
  
  const fontFamilies = new Set();
  for (const el of document.querySelectorAll('*')) {
    const fontFamily = getComputedStyle(el).fontFamily;
    // 解析字体家族
    const fonts = fontFamily.split(',').map(f => f.trim().toLowerCase());
    fontFamilies.add(fonts[0]);
  }
  
  // 移除通用字体
  const customFonts = [...fontFamilies].filter(f => !GENERIC_FONTS.has(f));
  
  if (customFonts.length === 1) {
    findings.push({
      id: 'single-font',
      snippet: `only "${customFonts[0]}" used on page`
    });
  }
  
  // 检测过度使用的字体
  for (const font of customFonts) {
    if (OVERUSED_FONTS.has(font) && !isBrandFontOnOwnDomain(font)) {
      findings.push({
        id: 'overused-font',
        snippet: `"${font}" is an overused font`
      });
    }
  }
  
  return findings;
}
```

## 八、错误处理

### 8.1 常见错误模式

| 错误 | 原因 | 解决方案 |
|------|------|----------|
| jsdom 返回 0×0 | 无布局 | 使用 `parseFloat(style.width)` |
| 背景简写未分解析 | jsdom 不解析 `background:` | 使用 `resolveBackground()` |
| 颜色格式不一致 | 浏览器标准化，jsdom 不 | 使用 `parseGradientColors()` |
| 假阳性 | 规则太宽泛 | 在夹具中添加 pass 案例 |

### 8.2 防御性编程
