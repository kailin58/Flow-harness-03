# ai-website-cloner 优秀模式借鉴推演验证

> 目标：在**零破坏**前提下，借鉴 ai-website-cloner-template 的优秀设计模式
> 原则：增量添加、功能开关、向后兼容、可回滚

---

## 一、现有架构分析

### 1.1 已有的 Worktree 机制

Flow Harness 已有 `SandboxManager` 支持 Git Worktree，但定位不同：

| 维度 | Flow Harness (现有) | ai-website-cloner |
|------|---------------------|-------------------|
| **目的** | 执行隔离（安全） | 并行构建（效率） |
| **创建时机** | 单任务执行前 | 多任务并行前 |
| **生命周期** | 任务完成后销毁 | 任务完成后合并 |
| **数量限制** | 最多 5 个 | 按需创建 |
| **结果处理** | 直接丢弃 | 合并到主分支 |

**结论**: 现有机制可复用，但需要扩展支持"并行+合并"模式。

### 1.2 已有的知识沉淀机制

Flow Harness 已有 `KnowledgeBase` 记录模式，但缺少结构化 spec 文件：

| 维度 | Flow Harness (现有) | ai-website-cloner |
|------|---------------------|-------------------|
| **格式** | JSON 内嵌 | 独立 spec 文件 |
| **内容** | 统计数据 | 详细的组件规范 |
| **可读性** | 机器可读 | 人机双可读 |
| **复用性** | 低 | 高 |

**结论**: 可扩展 KnowledgeBase 支持 spec 文件输出。

### 1.3 已有的流水线机制

Flow Harness 的 Supervisor 6 步闭环是串行的，缺少阶段门控：

| 维度 | Flow Harness (现有) | ai-website-cloner |
|------|---------------------|-------------------|
| **模式** | 串行闭环 | 流水线 + 阶段门控 |
| **门控** | 仅 Step 5 检查 | 每阶段都有门控 |
| **并行** | 支持（TaskDispatcher） | 强制并行 |
| **质量** | 评分阈值 | build 必须通过 |

**结论**: 可增加可选的"流水线模式"，不破坏现有 6 步闭环。

### 方案 1: 并行执行器 (ParallelExecutor)

**借鉴点**: ai-website-cloner 的 Worktree 并行构建模式

**设计原则**:

#### 2.1.1 新增模块设计

```javascript
// src/parallel-executor.js (新增文件)

/**
 * 并行执行器 - 借鉴 ai-website-cloner 的并行构建模式
 * 
 * 与现有 SandboxManager 的关系：
 * - SandboxManager: 用于安全隔离（单个任务）
 * - ParallelExecutor: 用于效率提升（多个任务并行）
 * 
 * 两者可以组合使用
 */
class ParallelExecutor {
  constructor(config = {}) {
    this.maxParallel = config.maxParallel || 4;
    this.mergeStrategy = config.mergeStrategy || 'auto'; // auto | manual | abort
    this.conflictResolver = config.conflictResolver || null;
    this.enabled = config.enabled !== false; // 默认启用，但需要显式调用
    
    // 不依赖 SandboxManager，保持独立
  }

  /**
   * 并行执行多个任务
   * @param {Array} tasks - 任务列表
   * @param {Object} executor - 执行器（通常是 AgentExecutor）
   * @returns {Promise<Object>} 执行结果
   */
  async executeParallel(tasks, executor, context = {}) {
    // 1. 创建 worktrees
    const worktrees = await this.createWorktrees(tasks);
    
    // 2. 并行执行
    const results = await Promise.all(
      tasks.map((task, index) => 
        executor.execute(task.agentId, task, {
          ...context,
          workingDir: worktrees[index].path,
          _parallelIndex: index
        })
      )
    );
    
    // 3. 合并结果
    const mergeResult = await this.mergeWorktrees(worktrees);
    
    // 4. 清理 worktrees
    await this.cleanupWorktrees(worktrees);
    
    return {
      results,
      mergeResult,
      success: results.every(r => r.success) && mergeResult.success
    };
  }

  // ... 其他方法
}

module.exports = ParallelExecutor;
```

```javascript
// supervisor-agent.js 修改方案（增量添加，不修改现有方法）

class SupervisorAgent {
  constructor(config) {
    // ... 现有代码保持不变 ...
    
    // 新增：可选的并行执行器（惰性初始化）
    this._parallelExecutor = null;
  }

  /**
   * 获取并行执行器（惰性初始化）
   * 只有在需要时才创建，不影响现有流程
   */
  getParallelExecutor() {
    if (!this._parallelExecutor) {
      const ParallelExecutor = require('./parallel-executor');
      this._parallelExecutor = new ParallelExecutor({
        maxParallel: this.config.maxParallelTasks || 4,
        enabled: this.config.enableParallelExecution || false
      });
    }
    return this._parallelExecutor;
  }

  /**
   * Step 4 执行 - 扩展版本（不修改原有方法）
   * 原有的 step4_execute 保持不变，新增此方法
   */
  async step4_execute_parallel(assignment, context = {}) {
    // 检查是否启用并行模式
    if (!this.config.enableParallelExecution) {
      // 未启用，回退到原有方法
      return this.step4_execute(assignment);
    }

    // 并行执行模式
    const parallelTasks = assignment.assignments
      .filter(a => a.subtask.constraints?.canRunInParallel)
      .map(a => ({
        agentId: this.mapExecutorToAgentId(a.executor.name),
        ...a.subtask
      }));

    if (parallelTasks.length < 2) {
      // 不足 2 个并行任务，回退到原有方法
      return this.step4_execute(assignment);
    }

    // 使用并行执行器
    const executor = this.getParallelExecutor();
    return await executor.executeParallel(parallelTasks, this.agentExecutor, context);
  }
}
```

#### 2.2.1 KnowledgeBase 扩展设计

```javascript
// src/knowledge-base.js 扩展（在文件末尾添加）

/**
 * 扩展方法：写入 Spec 文件
 * 借鉴 ai-website-cloner 的组件 spec 设计
 */
class KnowledgeBase {
  // ... 现有代码保持不变 ...

  /**
   * 写入 Spec 文件（新增方法）
   * @param {string} specName - Spec 名称
   * @param {Object} spec - Spec 内容
   * @param {Object} options - 选项
   */
  writeSpec(specName, spec, options = {}) {
    const specDir = path.join(this.knowledgePath, 'specs');
    
    // 确保目录存在
    if (!fs.existsSync(specDir)) {
      fs.mkdirSync(specDir, { recursive: true });
    }

    const specPath = path.join(specDir, `${specName}.json`);
    
    const specData = {
      name: specName,
      version: '1.0',
      createdAt: new Date().toISOString(),
      spec: spec,
      metadata: {
        taskType: options.taskType || 'unknown',
        successRate: options.successRate || null,
        reusability: this.calculateReusability(spec)
      }
    };

    fs.writeFileSync(specPath, JSON.stringify(specData, null, 2), 'utf8');
    
    return specPath;
  }

  /**
   * 读取 Spec 文件（新增方法）
   * @param {string} specName - Spec 名称
   * @returns {Object|null} Spec 内容
   */
  readSpec(specName) {
    const specPath = path.join(this.knowledgePath, 'specs', `${specName}.json`);
    
    if (!fs.existsSync(specPath)) {
      return null;
    }
    
    return JSON.parse(fs.readFileSync(specPath, 'utf8'));
  }

  /**
   * 列出所有 Spec（新增方法）
   * @returns {Array} Spec 列表
   */
  listSpecs() {
    const specDir = path.join(this.knowledgePath, 'specs');
    
    if (!fs.existsSync(specDir)) {
      return [];
    }
    
    return fs.readdirSync(specDir)
      .filter(f => f.endsWith('.json'))
      .map(f => {
        const spec = JSON.parse(fs.readFileSync(path.join(specDir, f), 'utf8'));
        return {
          name: spec.name,
          taskType: spec.metadata?.taskType,
          reusability: spec.metadata?.reusability,
          createdAt: spec.createdAt
        };
      });
  }

  /**
   * 计算可复用性评分（新增方法）
   * @param {Object} spec - Spec 内容
   * @returns {number} 0-1 的可复用性评分
   */
  calculateReusability(spec) {
    let score = 0;
    
    // 有明确的输入输出定义 +0.3
    if (spec.inputs && spec.outputs) score += 0.3;
    
    // 有依赖说明 +0.2
    if (spec.dependencies) score += 0.2;
    
    // 有示例 +0.2
    if (spec.examples) score += 0.2;
    
    // 有验收标准 +0.3
    if (spec.acceptanceCriteria) score += 0.3;
    
    return Math.min(1, score);
  }
}

// 不修改 module.exports，保持向后兼容
```

**借鉴点**: ai-website-cloner 的多阶段流水线质量门控

#### 2.3.1 流水线执行器设计

```javascript
// src/pipeline-executor.js (新增文件)

/**
 * 流水线执行器 - 借鉴 ai-website-cloner 的多阶段流水线
 * 
 * 与 Supervisor 6 步闭环的关系：
 * - 6 步闭环：默认模式，适合通用任务
 * - 流水线模式：可选模式，适合确定性高的任务
 */

/**
 * 流水线阶段定义
 */
const PIPELINE_STAGES = {
  RECONNAISSANCE: {
    name: 'reconnaissance',
    description: '侦察阶段 - 收集信息',
    gates: ['info_complete']
  },
  FOUNDATION: {
    name: 'foundation',
    description: '基础阶段 - 建立基础',
    gates: ['build_pass']
  },
  COMPONENT_SPEC: {
    name: 'component_spec',
    description: '组件规范阶段 - 定义规范',
    gates: ['spec_validated']
  },
  BUILD: {
    name: 'build',
    description: '构建阶段 - 并行构建',
    gates: ['build_pass', 'typecheck_pass']
  },
  ASSEMBLY: {
    name: 'assembly',
    description: '组装阶段 - 合并验证',
    gates: ['merge_success', 'final_build_pass']
  }
};

class PipelineExecutor {
  constructor(config = {}) {
    this.stages = config.stages || Object.values(PIPELINE_STAGES);
    this.enableParallelBuild = config.enableParallelBuild !== false;
    this.stopOnGateFailure = config.stopOnGateFailure !== false;
    
    // 引用外部组件（不创建新实例）
    this.agentExecutor = config.agentExecutor;
    this.knowledgeBase = config.knowledgeBase;
  }

  /**
   * 执行流水线
   * @param {Object} context - 执行上下文
   * @returns {Promise<Object>} 执行结果
   */
  async execute(context = {}) {
    const results = {
      stages: [],
      passed: true,
      failedAt: null
    };

    for (const stage of this.stages) {
      const stageResult = await this.executeStage(stage, context);
      results.stages.push(stageResult);

      // 门控检查
      if (!stageResult.passed) {
        results.passed = false;
        results.failedAt = stage.name;
        
        if (this.stopOnGateFailure) {
          break;
        }
      }
    }

    return results;
  }

  /**
   * 执行单个阶段
   */
  async executeStage(stage, context) {
    const result = {
      name: stage.name,
      passed: false,
      gates: {},
      output: null
    };

    // 执行阶段任务
    result.output = await this.runStageTasks(stage, context);

    // 执行门控检查
    for (const gate of stage.gates) {
      result.gates[gate] = await this.checkGate(gate, result.output);
    }

    // 综合判断
    result.passed = Object.values(result.gates).every(g => g.passed);

    return result;
  }

  /**
   * 检查门控
   */
  async checkGate(gateName, output) {
    const gateCheckers = {
      'info_complete': () => output && Object.keys(output).length > 0,
      'build_pass': () => this.runCommand('npm run build'),
      'typecheck_pass': () => this.runCommand('npx tsc --noEmit'),
      'spec_validated': () => output && output.specs && output.specs.length > 0,
      'merge_success': () => output && output.merged === true,
      'final_build_pass': () => this.runCommand('npm run build')
    };

    const checker = gateCheckers[gateName];
    if (!checker) {
      return { passed: true, reason: 'Unknown gate, auto-pass' };
    }

    try {
      const passed = await checker();
      return { passed: !!passed, gate: gateName };
    } catch (error) {
      return { passed: false, gate: gateName, error: error.message };
    }
  }

  /**
   * 运行命令
   */
  async runCommand(command) {
    const { execSync } = require('child_process');
    try {
      execSync(command, { stdio: 'pipe' });
      return true;
    } catch {
      return false;
    }
  }

  /**
   * 运行阶段任务
   */
  async runStageTasks(stage, context) {
    // 默认实现：返回空对象
    // 具体阶段由子类或配置覆盖
    return {};
  }
}

module.exports = { PipelineExecutor, PIPELINE_STAGES };
```

```javascript
// supervisor-agent.js 扩展（增量添加）

class SupervisorAgent {
  constructor(config) {
    // ... 现有代码不变 ...
    
    // 新增：执行模式
    this.executionMode = config.executionMode || 'closed_loop'; // closed_loop | pipeline
  }

  /**
   * 主入口 - 扩展支持模式选择
   * 不修改现有 handleTask，新增 handleTask_v2
   */
  async handleTask_v2(taskMessage, context = {}) {
    if (this.executionMode === 'pipeline') {
      return this.handleTask_pipeline(taskMessage, context);
    }
    // 默认使用原有 6 步闭环
    return this.handleTask(taskMessage, context);
  }

  /**
   * 流水线模式处理（新增方法）
   */
  async handleTask_pipeline(taskMessage, context = {}) {
    this.logger.info('\n🎯 Supervisor Agent 启动 (Pipeline 模式)');
    this.logger.info(`📝 任务: ${taskMessage}\n`);

    const { PipelineExecutor, PIPELINE_STAGES } = require('./pipeline-executor');
    
    const pipeline = new PipelineExecutor({
      agentExecutor: this.agentExecutor,
      knowledgeBase: this.knowledgeBase,
      enableParallelBuild: this.config.enableParallelBuild !== false
    });

    const result = await pipeline.execute({
      task: taskMessage,
      ...context
    });

    return {
      success: result.passed,
      mode: 'pipeline',
      stages: result.stages,
      failedAt: result.failedAt
    };
  }
}
```

```javascript
async handleTask_v2(taskMessage, context = {}) {
  try {
    if (this.executionMode === 'pipeline') {
      return await this.handleTask_pipeline(taskMessage, context);
    }
    return await this.handleTask(taskMessage, context);
  } catch (error) {
    // 流水线模式失败，自动降级到闭环模式
    if (this.executionMode === 'pipeline') {
      this.logger.warn('Pipeline mode failed, falling back to closed_loop');
      this.executionMode = 'closed_loop';
      return await this.handleTask(taskMessage, context);
    }
    throw error;
  }
}
```

## 六、总结

### 借鉴方案汇总

| 方案 | 借鉴内容 | 实现方式 | 破坏风险 |
|------|----------|----------|----------|
| ParallelExecutor | Worktree 并行构建 | 新增独立模块 | 无 |
| Spec 扩展 | Spec 文件沉淀 | 扩展 KnowledgeBase | 无 |
| PipelineExecutor | 流水线质量门控 | 新增独立模块 | 无 |

### 关键设计原则

1. **增量添加**: 所有新功能都是新增模块或方法，不修改现有逻辑
2. **配置开关**: 所有新功能默认禁用，需要显式启用
3. **向后兼容**: 新方法（如 `handleTask_v2`）与原方法共存
4. **可回滚**: 删除新增文件即可完全回滚
5. **降级机制**: 新功能失败自动降级到原有逻辑
